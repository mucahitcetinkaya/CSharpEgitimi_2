/*

/*
 
while de herhangi bir sart olmadıgı için degisken olusturduk dısarıda
su anda kod yarım yanlıs olanı görmek için boyle yapıyoruz

int sayac = 1;
while (sayac <= 10)
{
    Console.WriteLine(sayac);

}
Console.ReadLine();

sayac 1 içeri girdi 10 a kucuk esit true olur 
bunu görmek istiyorsan eger
breakpoint ekle calısırken
sayac <= 10 u sec sag tık add watch a tıkla görürsün
işlemin value ve type degerini verecektir.
value su true yazıyor mesela

devam edelim içeri girdi sayac 1 
10 dan kucuk esit girdi içine sayac yani 1 yazdırdı
tekrar tekrar bu işlemi yapacak 
artan bişey yok hep 1 
while sadece true false a baktıgı için siz bunu sarta baglı mı degiskenle mi sayacla mı sagladınız buna bakmıyor
sadece true ve false a bakıyor yoluna devam ediyor
yani biz kendimiz arttırmamız gerekiyor
Console.WriteLine(sayac); dan sonra
sayac++; ekledik yani 1 er arttırmıs olduk
int sayac = 1;
while (sayac <= 10)
{
    Console.WriteLine(sayac);
    sayac++;

}

zaten biz while dongusunu 1 den 10 na kadar olan sayıları göstermek için kullanmayız
işlemin sonucu false olana kadar devam etsin diye kullanırız
breakpoint le devam edelim sıra sıra devam edecek
11 olunca false olup cıkacak
add watch kısmında true olup false olarak degistiğinide görebilirsin
sayac degerinide görmek istersen eger ona da add watch diyebilirsin

sonuc olarak false olana kadar 
veya
break gorene kadar devam eder

 
 */

while (sayac <= 10)

    Console.WriteLine(sayac);
